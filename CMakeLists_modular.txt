# CMakeLists.txt for Modular Mini SONiC
cmake_minimum_required(VERSION 3.28)
project(MiniSonic_Modular VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard to C++20 with modules
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Enable C++20 modules support
if(CMAKE_VERSION VERSION_LESS "3.28")
    message(FATAL_ERROR "CMake 3.28 or higher is required for C++20 modules")
endif()

# Compiler-specific options
if(MSVC)
    add_compile_options(/std:c++20 /permissive- /Zc:__cplusplus /experimental:module)
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
else()
    add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find Boost (optional)
find_package(Boost 1.81 QUIET COMPONENTS system thread)
if(Boost_FOUND)
    message(STATUS "Found Boost ${Boost_VERSION}")
    add_compile_definitions(HAS_BOOST_ASIO)
else()
    message(STATUS "Boost not found - using fallback networking")
endif()

# Include directories
include_directories(src)
if(Boost_FOUND)
    include_directories(src ${Boost_INCLUDE_DIRS})
endif()

# Enable testing
enable_testing()

# Fetch Google Benchmark
include(FetchContent)
FetchContent_Declare(
    googlebenchmark
    URL https://github.com/google/benchmark/archive/refs/tags/v1.8.3.zip
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
set(BENCHMARK_ENABLE_TESTING OFF CACHE BOOL "Disable benchmark testing" FORCE)
FetchContent_MakeAvailable(googlebenchmark)

# Modular build configuration
set(MODULE_SOURCES
    src/modules/networking.ixx
    src/modules/networking.cpp
    src/modules/dataplane.ixx
    src/modules/dataplane.cpp
    src/modules/sai.ixx
    src/modules/sai.cpp
    src/modules/l2l3.ixx
    src/modules/l2l3.cpp
    src/modules/utils.ixx
    src/modules/utils.cpp
    src/modules/app.ixx
    src/modules/app.cpp
)

# Ensure .ixx files are treated as C++ sources
set_source_files_properties(${MODULE_SOURCES} PROPERTIES LANGUAGE CXX)

# Create modular executable
add_executable(mini_sonic_modular
    src/main_modular.cpp
    ${MODULE_SOURCES}
)

# Set module properties
set_target_properties(mini_sonic_modular PROPERTIES
    CXX_STANDARD 20
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
    CXX_SCAN_FOR_MODULES ON
)

# Link libraries
find_package(Threads REQUIRED)
target_link_libraries(mini_sonic_modular
    Threads::Threads
)

if(Boost_FOUND)
    target_link_libraries(mini_sonic_modular ${Boost_LIBRARIES})
endif()

# Modular benchmarks
add_executable(mini_sonic_modular_benchmarks
    benchmarks/main_modular_benchmark.cpp
    ${MODULE_SOURCES}
)

target_link_libraries(mini_sonic_modular_benchmarks
    benchmark
    benchmark_main
    Threads::Threads
)

if(Boost_FOUND)
    target_link_libraries(mini_sonic_modular_benchmarks ${Boost_LIBRARIES})
endif()

# Set benchmark properties
set_target_properties(mini_sonic_modular_benchmarks PROPERTIES
    CXX_STANDARD 20
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
    CXX_SCAN_FOR_MODULES ON
)

# Add include directories for benchmarks
target_include_directories(mini_sonic_modular_benchmarks PRIVATE src)

# Installation
install(TARGETS mini_sonic_modular
    RUNTIME DESTINATION bin
)

install(TARGETS mini_sonic_modular_benchmarks
    RUNTIME DESTINATION bin
    OPTIONAL
)

# Documentation
find_package(Doxygen QUIET)
if(DOXYGEN_FOUND)
    set(DOXYGEN_INPUT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/src/modules)
    set(DOXYGEN_OUTPUT_DIR ${CMAKE_CURRENT_BINARY_DIR}/docs)
    
    configure_file(${CMAKE_CURRENT_SOURCE_DIR}/docs/Doxyfile_modular.in
                   ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
    
    add_custom_target(docs_modular
        ${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
        WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
        COMMENT "Generating modular documentation with Doxygen"
        VERBATIM
    )
endif()

# Testing
enable_testing()
add_test(NAME ModularAppTest 
         COMMAND mini_sonic_modular --help
         WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})

# Print configuration summary
message(STATUS "")
message(STATUS "=== Mini SONiC Modular Configuration ===")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "C++ Modules: ENABLED")
message(STATUS "Boost Support: ${Boost_FOUND}")
message(STATUS "Build Type: ${CMAKE_BUILD_TYPE}")
message(STATUS "Compiler: ${CMAKE_CXX_COMPILER_ID}")
message(STATUS "==========================================")
message(STATUS "")
